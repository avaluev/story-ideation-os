## bootstrap: investor first-run setup (installs deps, prompts for OpenRouter key, runs doctor)
bootstrap:
	@bash scripts/bootstrap_investor.sh

## doctor: validate Python, uv, .env, and OpenRouter key connectivity
doctor:
	@uv run python scripts/preflight.py

## first-run: complete onboarding — bootstrap, then prompt for theme and run
first-run: bootstrap
	@echo
	@echo "What is your one-sentence story theme?"
	@read -r theme; \
	  if [ -z "$$theme" ]; then echo "No theme entered. Aborting."; exit 1; fi; \
	  echo; echo "Running pipeline with theme: $$theme"; echo; \
	  $(MAKE) single THEME="$$theme"

## package: build a scrubbed investor handoff tarball at dist/anomaly-engine-investor-v1.tar.gz
package:
	@bash scripts/build_handoff_tarball.sh

## investor-pack: build bilingual EN+RU investor markdown from the last completed run
investor-pack:
	@LATEST=$$(ls -td runs/*/ 2>/dev/null | head -1 | sed 's:/$$::'); \
	if [ -z "$$LATEST" ]; then echo "No runs/ directory found. Run 'make single' first."; exit 1; fi; \
	RUN_ID=$$(basename "$$LATEST"); \
	echo "Building investor pack for run $$RUN_ID..."; \
	uv run python scripts/build_investor_pack.py --run-id "$$RUN_ID"

# ---- SNIPPET END -----------------------------------------------------------
